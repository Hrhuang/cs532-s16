        function TealiumData() {
            
            this.utagData = {};
            
            /**
             * configure the utag_data variable to be passed on to Tealium
             */
            this.configure = function() {
                try {
                    var tealiumUser = new TealiumUser();

                    ea_origin_user_data = window.ea_origin_user;
                   
                    tealiumUser.configure(ea_origin_user_data);
            
                    this.utagData = {
                             userid : tealiumUser.getUserId(),

                             referring_site: "",
                             userstatus: tealiumUser.getLoginStatus(),
                            
                            ga_event_category : tealiumUser.event.category,
                            ga_event_action : tealiumUser.event.action,
                            ga_event_label : tealiumUser.event.label,
                            ga_event_value : tealiumUser.event.value
                    };
                } catch(e) {}
            }; 
            
            /**
             * getter method
             */
            this.getData = function() {
                this.configure();
                return this.utagData;
            };
        }

         /**
          * TealiumUserClass
          * @returns {TealiumUserEvent}
          */
         function TealiumUser() {
             
             this.user = null;
             
             /**
              * configure the class for the current user data
              */
             this.configure = function(userData) {
                 try {
                     this.setUser(userData);
                     this.setUserId();
                     this.setEvent();
                 } catch(e) { alert ('configure error ' + e);}
             };

             /**
              * parse the user data
              * expected format example: {"userId":"123456789","dateCreated":"2012-04-01T18:39Z","email":"me@ea.com"}
              * @param object userData
              */
             this.setUser = function(userData) {
               this.user = userData;  //since we changed the json to wrap the expecteformat inside a "data" object               
             };

             /**
              * writes the nucleus id to session storage
              * there is a case for a logged out event where we need the user id
              * but will no longer have it in the user object
              * so we need to store the value and reuse the stored values
              */
             this.setUserId = function() {
                 try {
                     if(this.user.pidId) {
                         sessionStorage['tlm_user_id'] = this.user.pidId;
                     }
                 } catch(e) {}
             };
             
             /**
              * gets the nucleus id from session storage
              * @returns
              */
             this.getUserId = function() {
                 try {
                     return sessionStorage['tlm_user_id'];
                 } catch(e) {
                     return '';
                 }
             };
             
             /**
              * clears the nucleus id from session storage
              */
             this.clearUserId = function() {
                 try {
                     sessionStorage['tlm_user_id'] = '';
                 } catch(e) {}
             };
             
             /**
              * get the current google event
              * @returns {TealiumUserEvent}
              */
             this.setEvent = function() {
                 // set a default (blank) event
                 var event = new TealiumUserEvent();
                 // evaluate if we have an event override
//                 if(this.getUserId()) {
                         // if we accessToken then we are logging in
                         if(typeof accessToken != 'undefined') {
                             // is this a new registration
                             if(this.isNewUser()) {
                                 event.register(this.getUserId());
                             } else {
                                 event.login(this.getUserId());
                             }
                         } else {
                             // if we are redirected by origin and do not have an accessToken but we have a user id then we are logging out
//                             if(this.getUserId()) {
                                 event.logout(this.getUserId());
//                                 this.clearUserId();
//                             }
                         }
//               }
                 this.event = event;
             };

             /**
              * get the current user login status
              * @returns
              */
             this.getLoginStatus = function() {
                 return this.getUserId() ? 'LoggedIn' : 'LoggedOut';
             };
             
             
             /**
              * a new acquisition (registration) will be a login that happens within a 60 seconds of the account being created
              * this is really hokey, but it's the only information we get back from origin
              * WARNING: getTime() calculates in milliseconds
              * @returns {Boolean}
              */
             this.isNewUser = function() {
                 try {
                     if(this.user.dateCreated) {
                         // the created datetime is in GMT
                         var created = new Date(this.user.dateCreated);
                         var gmtString = new Date().toISOString();
                         var current = new Date(gmtString);
                         var test = current.getTime() <= (created.getTime() + (120 * 1000));  //changed to 2 mins as origin is so slow 
                         return test;
                     }
                 } catch(e) {}
                 return false;
             };  

         };

         /**
          * TealiumUserEventClass
          * @returns {TealiumUserEvent}
          */
         function TealiumUserEvent() {
             
             this.category = '';
             this.action = '';
             this.label = '';
             this.value = '';
             

             /**
              * new registration event
              * @param userId
              */ 
             this.register = function(userId) {
                 this.setEvent('Acquisition', 'Registration', userId, '');
             };
                 
             /**
              * logged in event
              * @param userId
              */
             this.login = function(userId) {
                 this.setEvent('Userstatus', 'LoggedIn', userId, '');
             };
             
             /**
              * logged out event
              * @param userId
              */
             this.logout = function(userId) {
                 this.setEvent('Userstatus', 'LoggedOut', userId, '');
             };      
             
             /**
              * set a Google Analytics event
              * @param category
              * @param action
              * @param label
              * @param value
              */
             this.setEvent = function(category, action, label, value) {
                 this.category = category;
                 this.action = action;
                 this.label = label;
                 this.value = value;
             };  

         }
