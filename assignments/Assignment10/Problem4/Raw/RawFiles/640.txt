function url_with_next(url) {



    url = url.replace(/\?next=?&amp;/, '?').replace(/&amp;next=?&amp;/, '&amp;').replace(/[?&amp;]next=?$/, '').replace(/[?&amp;]next=?#/, '#');

    var next_val = window.location.pathname + window.location.search + window.location.hash;
    return url + connector + 'next=' + encodeURIComponent(next_val);
}
